Universal links

This plugin allows you to get the link with universal links(iOS 9.0+)

Runtime Events:

name:"universalLinks"

url: url opened with app

type: "cold", "warm"

Runtime:addEventListener("universalLinks", myListener)

Setup:

Add the following to your build.settings

settings =

{

	iphone =

	{	

		entitlements = {

			["com.apple.developer.associated-domains"] = {

				"applinks:www.mysite.com"

			},

		},		

	},	

	plugins={

		["plugin.universalLinks"] =

		{

			publisherId="tech.scotth",

			marketplaceId = "insert marketplace account ID",

		},

	},

}

Make sure to add the apple-app-site-association to html (check out apples site for more details)

{

	"applinks": {

		"apps": [],

		"details": [

		{

		"appID": "REPLACE.with.your.appID",

		"paths": ["*"]

		}

		]

	}

}
